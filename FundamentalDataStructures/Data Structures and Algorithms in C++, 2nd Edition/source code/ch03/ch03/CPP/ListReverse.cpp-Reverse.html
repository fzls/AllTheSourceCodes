  void listReverse(DLinkedList& L) {		// reverse a list
    DLinkedList T;				// temporary list
    while (!L.empty()) {			// reverse L into T
      string s = L.front();  L.removeFront();
      T.addFront(s);
    }
    while (!T.empty()) {			// copy T back to L
      string s = T.front();  T.removeFront();
      L.addBack(s);
    }
  }
