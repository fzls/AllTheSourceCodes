  int height2(const Tree& T, const Position& p) {
    if (p.isExternal()) return 0;		// leaf has height 0
    int h = 0;
    PositionList ch = p.children();		// list of children
    for (Iterator q = ch.begin(); q != ch.end(); ++q)
      h = max(h, height2(T, *q));
    return 1 + h;				// 1 + max height of children
  }
