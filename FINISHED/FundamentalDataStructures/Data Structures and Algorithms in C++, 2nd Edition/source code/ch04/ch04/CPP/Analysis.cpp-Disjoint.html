  bool areDisjoint(const vector<int>& a, const vector<int>& b,
  		   const vector<int>& c) {
    for (int i = 0; i < a.size(); i++) 
      for (int j = 0; j < b.size(); j++) 
	for (int k = 0; k < c.size(); k++) 
	  if ((a[i] == b[j]) && (b[j] == c[k])) return false;
    return true;
  }
