  int height1(const Tree& T) {
    int h = 0;
    PositionList nodes = T.positions();		// list of all nodes
    for (Iterator q = nodes.begin(); q != nodes.end(); ++q) {
      if (q->isExternal())
        h = max(h, depth(T, *q));		// get max depth among leaves
    }
    return h;
  }
