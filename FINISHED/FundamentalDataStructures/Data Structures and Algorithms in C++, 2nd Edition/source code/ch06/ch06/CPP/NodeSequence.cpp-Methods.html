  						// get position from index
  NodeSequence::Iterator NodeSequence::atIndex(int i) const {
    Iterator p = begin();
    for (int j = 0; j < i; j++) ++p;
    return p;
  }
  						// get index from position
  int NodeSequence::indexOf(const Iterator& p) const {
    Iterator q = begin();
    int j = 0;
    while (q != p) {				// until finding p
      ++q; ++j;					// advance and count hops
    }
    return j;
  }
